29inline std::string trim(const std::string& str) {
30 size_t start = str.find_first_not_of(" \t\n\r");
31 if (start == std::string::npos) return "";
32 size_t end = str.find_last_not_of(" \t\n\r");
33 return str.substr(start, end - start + 1);
34}

113 Token next_token() {
114 skip_whitespace();
115
116 if (pos >= text.length()) {
117 return {.type = TokenType::EOF_TOKEN, .value = "", .position = pos};
118 }
119
120 size_t start_pos = pos;
121
122 // Check for operators and parentheses
123 auto is_boundary_char = [&](size_t idx) {
124 if (idx >= text.length()) {
125 return true;
126 }
127 char ch = text[idx];
128 unsigned char c = static_cast<unsigned char>(ch);
129 return std::isspace(c) || ch == '(' || ch == ')';
130 };
131 auto has_boundary_before = [&](size_t idx) {
132 if (idx == 0) {
133 return true;
134 }
135 char ch = text[idx - 1];
136 unsigned char c = static_cast<unsigned char>(ch);
137 return std::isspace(c) || ch == '(' || ch == ')';
138 };
139
140 if (pos + 3 <= text.length() && text.compare(pos, 3, "AND") == 0 && has_boundary_before(pos) &&
141 is_boundary_char(pos + 3)) {
142 pos += 3;
143 return {.type = TokenType::AND, .value = "AND", .position = start_pos};
144 }
145 if (pos + 2 <= text.length() && text.compare(pos, 2, "OR") == 0 && has_boundary_before(pos) &&
146 is_boundary_char(pos + 2)) {
147 pos += 2;
148 return {.type = TokenType::OR, .value = "OR", .position = start_pos};
149 }
150 if (pos + 3 <= text.length() && text.compare(pos, 3, "XOR") == 0 && has_boundary_before(pos) &&
151 is_boundary_char(pos + 3)) {
152 pos += 3;
153 return {.type = TokenType::XOR, .value = "XOR", .position = start_pos};
154 }
155 if (pos + 3 <= text.length() && text.compare(pos, 3, "NOT") == 0 && has_boundary_before(pos) &&
156 is_boundary_char(pos + 3)) {
157 pos += 3;
158 return {.type = TokenType::NOT, .value = "NOT", .position = start_pos};
159 }
160 if (text[pos] == '(') {
161 pos++;
162 return {.type = TokenType::LPAREN, .value = "(", .position = start_pos};
163 }
164 if (text[pos] == ')') {
165 pos++;
166 return {.type = TokenType::RPAREN, .value = ")", .position = start_pos};
167 }
168
169 // Parse variable name (any non-whitespace that's not an operator or parentheses)
170 if (!std::isspace(static_cast<unsigned char>(text[pos])) && text[pos] != '(' &&
171 text[pos] != ')') {
172 std::string var_name;
173 while (pos < text.length() && !std::isspace(static_cast<unsigned char>(text[pos])) &&
174 text[pos] != '(' && text[pos] != ')') {
175 var_name += text[pos];
176 pos++;
177 }
178
179 if (!var_name.empty()) {
180 return {.type = TokenType::VARIABLE, .value = var_name, .position = start_pos};
181 }
182 }
183
184 throw std::runtime_error(
185 std::format("Unexpected character at position {}: '{}'", pos, text[pos]));
186 }

247 my_expression_ptr parse_primary() {
248 if (current_token.type == Tokenizer::TokenType::VARIABLE) {
249 std::string var_name = current_token.value;
250 advance();
251 return std::make_unique<my_expression>(my_variable{var_name});
252 } else if (current_token.type == Tokenizer::TokenType::LPAREN) {
253 advance(); // consume '('
254 auto expr = parse_expression();
255 expect(Tokenizer::TokenType::RPAREN);
256 return expr;
257 } else {
258 throw std::runtime_error(
259 std::format("Expected variable or '(' at position {}", current_token.position));
260 }
261 }

271 my_expression_ptr parse_not_expr() {
272 if (current_token.type == Tokenizer::TokenType::NOT) {
273 advance(); // consume 'NOT'
274 auto operand = parse_not_expr(); // right-associative
275 return std::make_unique<my_expression>(my_not{std::move(operand)});
276 } else {
277 return parse_primary();
278 }
279 }

289 my_expression_ptr parse_and_expr() {
290 auto left = parse_not_expr();

292 while (current_token.type == Tokenizer::TokenType::AND) {
293 advance(); // consume 'AND'

295 left = std::make_unique<my_expression>(my_and{std::move(left), std::move(right)});

388inline my_expression_ptr parse_expression(const std::string& expr_str) {
389 std::string trimmed = trim(expr_str);

391 if (trimmed.empty()) {

428inline my_expression_ptr read_expression_from_file(const std::string& filename) {
429 std::ifstream file(filename);
430 if (!file.is_open()) {
431 throw std::runtime_error("Could not open file: " + filename);
432 }
433
434 std::string line;
435 std::string expression_str;
436
437 // Read all non-empty lines and concatenate them
438 while (std::getline(file, line)) {
439 line = trim(line);
440 if (!line.empty() && line[0] != '#') { // Skip empty lines and comments
441 expression_str += " " + line;
442 }
443 }
444
445 expression_str = trim(expression_str);
446 if (expression_str.empty()) {
447 throw std::runtime_error("No expression found in file: " + filename);
448 }
449
450 return parse_expression(expression_str);
451}
