41 dagir::ir_attr_map out;

85struct expression_edge_attributor {
86 using handle = typename expression_read_only_dag_view::handle;
87
88 dagir::ir_attr_map operator()(const expression_read_only_dag_view& /*view*/, const handle& parent,
89 const handle& child) const {
90 dagir::ir_attr_map out;
91 if (!parent.ptr) return out;
92 // For binary operators, label edges left/right. Keep some styles
93 // (e.g. bold for AND, dashed for NOT) but do not set edge colors.
94 if (auto p_and = std::get_if<my_and>(parent.ptr)) {
95 if (p_and->left && p_and->left.get() == child.ptr)
96 out.emplace(dagir::ir_attrs::k_label, std::string{"L"});
97 else if (p_and->right && p_and->right.get() == child.ptr)
98 out.emplace(dagir::ir_attrs::k_label, std::string{"R"});
99 } else if (auto p_or = std::get_if<my_or>(parent.ptr)) {
100 if (p_or->left && p_or->left.get() == child.ptr)
101 out.emplace(dagir::ir_attrs::k_label, std::string{"L"});
102 else if (p_or->right && p_or->right.get() == child.ptr)
103 out.emplace(dagir::ir_attrs::k_label, std::string{"R"});
104 } else if (auto p_xor = std::get_if<my_xor>(parent.ptr)) {
105 if (p_xor->left && p_xor->left.get() == child.ptr)
106 out.emplace(dagir::ir_attrs::k_label, std::string{"L"});
107 else if (p_xor->right && p_xor->right.get() == child.ptr)
108 out.emplace(dagir::ir_attrs::k_label, std::string{"R"});
109 } else if (auto p_not = std::get_if<my_not>(parent.ptr)) {
110 // Unary operator: do not emit edge labels for NOT -- leave edges unlabeled
111 (void)child;
112 } else if (std::get_if<my_variable>(parent.ptr)) {
113 // variables have no outgoing labeled edges
114 }
115
116 return out;
117 }
118};
