108template <dagir::concepts::read_only_dag_view View, class NodePolicy, class EdgePolicy>
109 requires dagir::concepts::node_attributor<NodePolicy, View>
110ir_graph build_ir(const View& view, NodePolicy&& node_policy, EdgePolicy&& edge_attr) {
111 using H = typename View::handle;
112 using key_t = std::uint64_t;
113
114 ir_graph graph;
115
116 // Get a deterministic traversal order (topological for DAGs). We traverse
117 // nodes in topological order and generate edges as we go.
118 std::vector<H> topo = kahn_topological_order(view);
119
120 graph.nodes.reserve(topo.size());
121
122 // First, create nodes (memoized) using label policy
123 for (std::size_t idx = 0; idx < topo.size(); ++idx) {
124 const H& h = topo[idx];
125 key_t k = h.stable_key();
126
127 // Optionally guard traversal for this node
128 if constexpr (requires(const View& v, H hh) { v.start_guard(hh); }) {
129 auto guard = view.start_guard(h);
130 (void)guard;
131 }
132
133 ir_node n;
134 n.id = k;
135
136 // Default canonical name assigned in topological order; policies must
137 // be node-attributors producing `dagir::ir_attr_map` that will populate
138 // `n.attributes`. We prefer attribute-provided values; otherwise the
139 // default name is used and a label from the stable key is written.
140 auto attributes = std::invoke(node_policy, view, h);
141
142 // Copy the returned attributes into the node. Note the type may not be directly compatible.
143 for (const auto& [attr_key, attr_value] : attributes) {
144 n.attributes[attr_key] = attr_value;
145 }
146
147 if (!n.attributes.count(ir_attrs::k_name))
148 n.attributes[ir_attrs::k_name] = std::format("node{:03}", idx);
149 if (!n.attributes.count(ir_attrs::k_label)) n.attributes[ir_attrs::k_label] = std::to_string(k);
150
151 graph.nodes.push_back(std::move(n));
152 }
153
154 // Now collect edges; reserve approximate size by summing child counts
155 // Reserve an approximate size for edges by summing child counts using
156 // standard algorithms. Using std::accumulate makes the intent clearer
157 // and satisfies cppcheck's `useStlAlgorithm` suggestion.
158 std::size_t est_edges = std::accumulate(
159 topo.begin(), topo.end(), std::size_t{0}, [&view](std::size_t acc, const H& h) {
160 return acc + static_cast<std::size_t>(std::ranges::distance(view.children(h)));
161 });
162 graph.edges.reserve(est_edges);
163
164 for (const H& parent : topo) {
165 key_t pk = parent.stable_key();
166 for (auto const& edge_like : view.children(parent)) {
167 H child = build_ir_extract_child<H>(edge_like);
168 key_t ck = child.stable_key();
169
170 ir_edge ie;
171 ie.source = pk;
172 ie.target = ck;
173
174 // Determine attributes via flexible invocation forms
175 if constexpr (std::invocable<EdgePolicy, const View&, const H&, const decltype(edge_like)&>) {
176 ie.attributes = std::invoke(edge_attr, view, parent, edge_like);
177 } else if constexpr (std::invocable<EdgePolicy, const View&, const H&, const H&>) {
178 ie.attributes = std::invoke(edge_attr, view, parent, child);
179 } else if constexpr (std::invocable<EdgePolicy, const H&, const decltype(edge_like)&>) {
180 ie.attributes = std::invoke(edge_attr, parent, edge_like);
181 } else if constexpr (std::invocable<EdgePolicy, const H&, const H&>) {
182 ie.attributes = std::invoke(edge_attr, parent, child);
183 } else {
184 ie.attributes = {};
185 }
186
187 graph.edges.push_back(std::move(ie));
188 }
189 }
190
191 return graph;
192}

200template <dagir::concepts::read_only_dag_view View>
201ir_graph build_ir(const View& view) {
202 auto node_attr = [](auto const& /*view*/, auto const& h) -> dagir::ir_attr_map {
203 dagir::ir_attr_map m;
204 m.emplace(ir_attrs::k_label, std::format("{}", h.stable_key()));
205 return m;
206 };
207 auto edge_attr = [](auto&&...) -> dagir::ir_attr_map { return {}; };
208 return build_ir(view, node_attr, edge_attr);
209}
