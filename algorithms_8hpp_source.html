41template <dagir::concepts::read_only_dag_view View>
42std::vector<typename View::handle> kahn_topological_order(const View& view) {
43 using H = typename View::handle;
44 using key_t = std::uint64_t;
45
46 std::unordered_map<key_t, std::size_t> indeg;
47 std::unordered_map<key_t, H> handle_of;
48 std::unordered_set<key_t> seen;
49 std::vector<key_t> nodes; // all discovered node keys (for iteration)
50
51 // helper to extract a child handle from a range element (edge or handle).
52 auto extract_child = []<class E>(const E& e) -> H {
53 if constexpr (std::convertible_to<E, H>) {
54 return static_cast<H>(e);
55 } else {
56 return e.target();
57 }
58 };
59
60 // BFS/stack from roots to discover reachable nodes and compute indegrees
61 std::vector<H> work;
62 for (auto const& r : view.roots()) {
63 H h = r;
64 key_t k = h.stable_key();
65 if (seen.insert(k).second) {
66 handle_of.emplace(k, h);
67 indeg.try_emplace(k, 0);
68 nodes.push_back(k);
69 work.push_back(h);
70 }
71 }
72
73 for (std::size_t i = 0; i < work.size(); ++i) {
74 H cur = work[i];
75 for (auto const& edge_like : view.children(cur)) {
76 H child = extract_child(edge_like);
77 key_t ck = child.stable_key();
78 // ensure child is in maps
79 handle_of.try_emplace(ck, child);
80 auto [it, inserted] = indeg.try_emplace(ck, 0);
81 ++(it->second);
82
83 if (seen.insert(ck).second) {
84 nodes.push_back(ck);
85 work.push_back(child);
86 }
87 }
88 }
89
90 // Kahn: push all zero-indegree nodes
91 std::queue<key_t> q;
92 for (auto const& k : nodes) {
93 if (indeg[k] == 0) q.push(k);
94 }
95
96 std::vector<H> order;
97 order.reserve(nodes.size());
98
99 while (!q.empty()) {
100 key_t k = q.front();
101 q.pop();
102 order.push_back(handle_of.at(k));
103
104 H h = handle_of.at(k);
105 for (auto const& edge_like : view.children(h)) {
106 H child = extract_child(edge_like);
107 key_t ck = child.stable_key();
108 auto it = indeg.find(ck);
109 if (it == indeg.end()) continue; // child outside discovered set
110 if (--(it->second) == 0) q.push(ck);
111 }
112 }
113
114 if (order.size() != nodes.size())
115 throw std::runtime_error("kahn_topological_order: cycle detected in reachable graph");
116
117 return order;
118}

138template <dagir::concepts::read_only_dag_view View, class R, class Combiner>
139auto postorder_fold(const View& view, Combiner combiner) -> std::unordered_map<std::uint64_t, R> {
140 using H = typename View::handle;
141 using key_t = std::uint64_t;
142
143 auto topo = kahn_topological_order(view);
144 std::unordered_map<key_t, R> results;
145 results.reserve(topo.size());
146
147 // helper to extract child handle (same as above)
148 auto extract_child = []<class E>(const E& e) -> H {
149 if constexpr (std::convertible_to<E, H>) {
150 return static_cast<H>(e);
151 } else {
152 return e.target();
153 }
154 };
155
156 // Process in reverse topological order: children before parents
157 for (auto it = topo.rbegin(); it != topo.rend(); ++it) {
158 H node = *it;
159 key_t k = node.stable_key();
160
161 // collect child results in order observed
162 std::vector<R> child_vals;
163 for (auto const& edge_like : view.children(node)) {
164 H child = extract_child(edge_like);
165 key_t ck = child.stable_key();
166 auto found = results.find(ck);
167 if (found != results.end())
168 child_vals.push_back(found->second);
169 else
170 child_vals.push_back(R{}); // absent->default constructed result
171 }
172
173 R res = std::invoke(combiner, view, node, std::span(child_vals));
174 results.emplace(k, std::move(res));
175 }
176
177 return results;
178}
