45 dagir::ir_attr_map operator()(const typename teddy_read_only_dag_view::handle& h) const {
46 dagir::ir_attr_map out;
47 if (!h.ptr) return out;
48
49 // Terminal nodes: show their boolean value
50 if (h.ptr->is_terminal()) {
51 // Ensure terminal nodes are labeled with either "0" or "1" explicitly.
52 const auto val = h.ptr->get_value();
53 out.emplace(dagir::ir_attrs::k_label, val ? std::string{"1"} : std::string{"0"});
54 out.emplace(dagir::ir_attrs::k_shape, std::string{"box"});
55 out.emplace(dagir::ir_attrs::k_fill_color, std::string{"lightgray"});
56 } else {
57 // Variable nodes: label with variable name if available, otherwise index
58 std::string label = std::to_string(h.ptr->get_index());
59 // view-aware overload will populate var_names via the view argument when available
60 out.emplace(dagir::ir_attrs::k_label, label);
61 out.emplace(dagir::ir_attrs::k_shape, std::string{"circle"});
62 }
63
64 return out;

70 dagir::ir_attr_map operator()(const teddy_read_only_dag_view& view,
71 const typename teddy_read_only_dag_view::handle& h) const {
72 dagir::ir_attr_map out = operator()(h);
73 if (!h.ptr) return out;
74
75 // If the view provided variable names, use them for variable nodes
76 const auto* names = view.var_names();
77 if (names && !h.ptr->is_terminal()) {
78 int idx = h.ptr->get_index();
79 if (idx >= 0 && static_cast<size_t>(idx) < names->size()) {
80 const std::string nm = (*names)[static_cast<size_t>(idx)];
81 out[dagir::ir_attrs::k_label] = nm;
82 }
83 }
84
85 // Always assign a unique renderer-visible id attribute derived from the
86 // node's stable key. This ensures distinct nodes receive distinct ids
87 // even when labels collide.
88 out[dagir::ir_attrs::k_id] = dagir::utility::make_node_id(h.stable_key());

115 // Determine whether this child is the false (0) or true (1) branch.
116 auto son0 = parent.ptr->get_son(0);
117 auto son1 = parent.ptr->get_son(1);
118
119 if (son0 && son0 == child.ptr) {
120 out.emplace(dagir::ir_attrs::k_style, std::string{"dashed"});
121 } else if (son1 && son1 == child.ptr) {
