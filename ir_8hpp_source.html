51inline bool operator<(ir_node const& a, ir_node const& b) {
52 const auto a_it = a.attributes.find(ir_attrs::k_name);
53 const auto b_it = b.attributes.find(ir_attrs::k_name);
54 const bool a_has = (a_it != a.attributes.end());
55 const bool b_has = (b_it != b.attributes.end());
56 if (a_has && b_has) {
57 const std::string& a_name = a_it->second;
58 const std::string& b_name = b_it->second;
59 if (a_name != b_name) return a_name < b_name;
60 return a.id < b.id;
61 }
62 if (a_has && !b_has) return true; // named nodes come before unnamed
63 if (!a_has && b_has) return false; // unnamed after named
64 return a.id < b.id;
65}
