47inline std::string escape_dot(const std::string& s) {
48 std::string out;
49 out.reserve(s.size() + 8);
50 for (size_t i = 0; i < s.size(); ++i) {
51 unsigned char c = static_cast<unsigned char>(s[i]);
52 switch (c) {
53 case '\\':
54 // Always escape backslash
55 out += "\\\\";
56 // If this backslash is followed by an escString letter, preserve the
57 // literal by leaving the following character as-is (we've already
58 // escaped the backslash itself).
59 break;
60 case '"':
61 out += "\\\"";
62 break;
63 case '\n':
64 out += "\\n";
65 break;
66 case '\r':
67 out += "\\r";
68 break;
69 case '\t':
70 out += "\\t";
71 break;
72 case '\f':
73 out += "\\f";
74 break;
75 case '\v':
76 out += "\\v";
77 break;
78 default:
79 if (c < 0x20) {
80 // Non-printable control characters: emit hex escape to avoid
81 // introducing invalid bytes in DOT source.
82 out += std::format("\\x{:02x}", static_cast<unsigned>(c));
83 } else {
84 out += static_cast<char>(c);
85 }
86 break;
87 }
88 }
89 return out;
90}

109// Writes a GraphViz DOT representation of `g` to `os`.
110// `graph_name` is used as the DOT graph identifier.
111inline void render_dot(std::ostream& os, const ir_graph& g, std::string_view graph_name = "G") {
112 os << "digraph " << graph_name << " {\n";
113
114 // Emit default rankdir only if the graph-level attributes do not provide one.
115 if (!g.global_attrs.count(ir_attrs::k_rankdir)) {
116 os << " rankdir=TB;\n"; // default top-to-bottom layout
117 }
118
119 // First, emit global graph attributes (map known keys) in lexicographic order
120 if (!g.global_attrs.empty()) {
121 std::vector<std::string_view> gkeys;
122 gkeys.reserve(g.global_attrs.size());
123 std::transform(g.global_attrs.begin(), g.global_attrs.end(), std::back_inserter(gkeys),
124 [](auto const& p) { return p.first; });
125 std::sort(gkeys.begin(), gkeys.end(), [](std::string_view a, std::string_view b) {
126 return std::string_view(a) < std::string_view(b);
127 });
128 for (const auto& k : gkeys) {
129 const auto& v = g.global_attrs.at(k);
130 if (k == ir_attrs::k_graph_label) {
131 os << " label=\"" << render_dot_detail::escape_dot(v) << "\";\n";
132 } else {
133 os << " " << k << "=\"" << render_dot_detail::escape_dot(v) << "\";\n";
134 }
135 }
136 }
137
138 std::unordered_map<std::uint64_t, std::string> name_map;
139 // Gather node names for use in edge emission.
140
141 // Emit nodes
142 for (const auto& n : g.nodes) {
143 // Attribute map from node
144 const auto& amap = n.attributes;
145
146 // Determine the node identifier: prefer attribute "name", then fall back
147 // to a generated id. We also populate a printable label via k_label.
148 // Prefer canonical `k_id` as the stable node identifier; for
149 // historical compatibility also accept a literal "name" attribute.
150 const bool has_explicit_name = amap.count(ir_attrs::k_id) || amap.count("name");
151 const std::string raw_node_name =
152 has_explicit_name ? (amap.count(ir_attrs::k_id) ? amap.at(ir_attrs::k_id) : amap.at("name"))
153 : std::format("n{}", n.id);
154 // If the node name was provided by a policy, escape and quote it so
155 // arbitrary strings remain valid DOT identifiers. If the renderer
156 // generated the name (n{id}), keep it unquoted to preserve the
157 // historical emission format (tests and tools often expect unquoted ids).
158 const std::string node_name =
159 has_explicit_name ? std::format("\"{}\"", render_dot_detail::escape_dot(raw_node_name))
160 : raw_node_name;
161 name_map[n.id] = node_name;
162
163 // Ensure label: prefer k_label, then generated id
164 std::string label =
165 amap.count(ir_attrs::k_label) ? amap.at(ir_attrs::k_label) : std::format("{}", n.id);
166
167 // Work from a local mutable copy when applying defaults so we don't mutate the
168 // const attribute map stored on the node.
169 auto local = amap;
170 if (!local.count(ir_attrs::k_style)) {
171 local[ir_attrs::k_style] = "filled";
172 }
173
174 // Emit node using the possibly-updated local map. Emit attributes in
175 // lexicographic order for deterministic output. Label is emitted first.
176 os << " " << node_name << " [";
177 os << "label = \"" << render_dot_detail::escape_dot(label) << "\"";
178 if (!local.empty()) {
179 std::vector<std::string_view> keys;
180 keys.reserve(local.size());
181 std::transform(local.begin(), local.end(), std::back_inserter(keys),
182 [](auto const& p) { return p.first; });
183 std::sort(keys.begin(), keys.end(),
184 [](std::string_view a, std::string_view b) { return a < b; });
185 for (const auto& k : keys) {
186 if (k == ir_attrs::k_label) continue;
187 if (k == ir_attrs::k_id) {
188 os << ", name = \"" << render_dot_detail::escape_dot(local.at(k)) << "\"";
189 continue;
190 }
191 // Avoid emitting a literal "name" attribute if we've already emitted
192 // the canonical id as `name` above. This prevents duplicate `name`
193 // attributes when both `k_id` and a historical `"name"` key exist.
194 if (k == ir_attrs::k_name) continue;
195 os << ", " << k << " = \"" << render_dot_detail::escape_dot(local.at(k)) << "\"";
196 }
197 }
198 os << "];\n";
199 }
200
201 // Emit edges (use previously computed name_map for identifiers)
202 for (const auto& e : g.edges) {
203 const std::string src = name_map.at(e.source);
204 const std::string dst = name_map.at(e.target);
205
206 const auto& amap = e.attributes;
207
208 os << " " << src << " -> " << dst;
209 if (!amap.empty()) {
210 os << " [";
211 bool first = true;
212 if (amap.count(ir_attrs::k_label)) {
213 os << "label = \"" << render_dot_detail::escape_dot(amap.at(ir_attrs::k_label)) << "\"";
214 first = false;
215 }
216 std::vector<std::string_view> keys;
217 keys.reserve(amap.size());
218 std::transform(amap.begin(), amap.end(), std::back_inserter(keys),
219 [](auto const& p) { return p.first; });
220 std::sort(keys.begin(), keys.end(),
221 [](std::string_view a, std::string_view b) { return a < b; });
222 for (const auto& k : keys) {
223 if (k == ir_attrs::k_label) continue;
224 if (!first) os << ", ";
225 first = false;
226 os << k << " = \"" << render_dot_detail::escape_dot(amap.at(k)) << "\"";
227 }
228 os << "]";
229 }
230 os << ";\n";
231 }
232
233 os << "}\n";
234}
