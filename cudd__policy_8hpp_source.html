34 dagir::ir_attr_map operator()(const typename cudd_read_only_dag_view::handle& h) const {
35 dagir::ir_attr_map out;
36 if (!h.ptr) return out;
37
38 if (Cudd_IsConstant(h.ptr)) {
39 // In CUDD constants are represented as (possibly) complemented pointers
40 // to the logical-one node. Use the complement flag to determine value.
41 const bool is_complement = Cudd_IsComplement(h.ptr);
42 int val = is_complement ? 0 : 1;
43 out.emplace(dagir::ir_attrs::k_label, val ? std::string{"1"} : std::string{"0"});
44 out.emplace(dagir::ir_attrs::k_shape, std::string{"box"});
45 out.emplace(dagir::ir_attrs::k_fill_color, std::string{"lightgray"});
46 } else {
47 DdNode* base = Cudd_Regular(h.ptr);
48 std::string label = std::to_string(Cudd_NodeReadIndex(base));
49 out.emplace(dagir::ir_attrs::k_label, label);
50 out.emplace(dagir::ir_attrs::k_shape, std::string{"circle"});
51 }
52
53 return out;
54 }

56 dagir::ir_attr_map operator()(const cudd_read_only_dag_view& view,
57 const typename cudd_read_only_dag_view::handle& h) const {
58 dagir::ir_attr_map out = operator()(h);
59 if (!h.ptr) return out;
60
61 const auto* names = view.var_names();
62 if (names && !Cudd_IsConstant(h.ptr)) {
63 DdNode* base = Cudd_Regular(h.ptr);
64 int idx = Cudd_NodeReadIndex(base);
65 if (idx >= 0 && static_cast<size_t>(idx) < names->size()) {
66 const std::string nm = (*names)[static_cast<size_t>(idx)];
67 out[dagir::ir_attrs::k_label] = nm;

69 }

71 // Assign unique node id attribute based on stable key

78struct cudd_edge_attributor {

81 dagir::ir_attr_map operator()(const cudd_read_only_dag_view& /*view*/, const handle& parent,
82 const handle& child) const {
83 dagir::ir_attr_map out;
84 if (!parent.ptr) return out;
85
86 const bool is_comp = Cudd_IsComplement(parent.ptr);
87 DdNode* base = Cudd_Regular(parent.ptr);
88 DdNode* then_child = Cudd_T(base);
89 DdNode* else_child = Cudd_E(base);
90 if (is_comp) {
91 if (else_child) else_child = Cudd_Not(else_child);
92 if (then_child) then_child = Cudd_Not(then_child);
93 }
94
95 if (else_child && else_child == child.ptr) {
96 out.emplace(dagir::ir_attrs::k_style, std::string{"dashed"});
97 } else if (then_child && then_child == child.ptr) {
98 out.emplace(dagir::ir_attrs::k_style, std::string{"solid"});
99 }
100
101 return out;
102 }
103};
