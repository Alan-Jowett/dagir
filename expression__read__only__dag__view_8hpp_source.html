25// Non-owning handle pointing at a node inside a parsed expression AST.
26struct expression_handle {
27 const my_expression* ptr = nullptr;
28
29 // Use non-constexpr stable_key since pointer-to-integer casts are not
30 // permitted in a constexpr context on all compilers.
31 std::uint64_t stable_key() const noexcept {
32 return static_cast<std::uint64_t>(reinterpret_cast<std::uintptr_t>(ptr));
33 }
34
35 constexpr const void* debug_address() const noexcept { return static_cast<const void*>(ptr); }
36
37 constexpr bool operator==(const expression_handle& o) const noexcept { return ptr == o.ptr; }
38 constexpr bool operator!=(const expression_handle& o) const noexcept { return ptr != o.ptr; }
39};

43class expression_read_only_dag_view {
44 public:
45 using handle = expression_handle;
46
47 // Lightweight edge type for this adapter satisfying `edge_ref` concept.
48 struct expression_edge {
49 handle to;
50 constexpr const handle& target() const noexcept { return to; }
51 };
52
53 explicit expression_read_only_dag_view(const my_expression* root = nullptr) : root_{root} {}
54
55 // Return a range (vector) of edges for the given handle. Edges carry target handles.
56 auto children(const handle& h) const {
57 std::vector<expression_edge> out;
58 if (!h.ptr) return out;
59
60 if (auto p_and = std::get_if<my_and>(h.ptr)) {
61 if (p_and->left) out.push_back(expression_edge{handle{p_and->left.get()}});
62 if (p_and->right) out.push_back(expression_edge{handle{p_and->right.get()}});
63 } else if (auto p_or = std::get_if<my_or>(h.ptr)) {
64 if (p_or->left) out.push_back(expression_edge{handle{p_or->left.get()}});
65 if (p_or->right) out.push_back(expression_edge{handle{p_or->right.get()}});
66 } else if (auto p_xor = std::get_if<my_xor>(h.ptr)) {
67 if (p_xor->left) out.push_back(expression_edge{handle{p_xor->left.get()}});
68 if (p_xor->right) out.push_back(expression_edge{handle{p_xor->right.get()}});
69 } else if (auto p_not = std::get_if<my_not>(h.ptr)) {
70 if (p_not->expr) out.push_back(expression_edge{handle{p_not->expr.get()}});
71 }
72
73 return out;
74 }
75
76 // Return roots as a one-element range (or empty if no root)
77 auto roots() const {
78 if (!root_) return std::vector<handle>{};
79 return std::vector<handle>{handle{root_}};
80 }
81
82 // No-op guard for this simple in-memory view
83 static auto start_guard(const handle&) { return dagir::noop_guard{}; }
84
85 private:
86 const my_expression* root_ = nullptr;
87};
