56 static auto children(const handle& h) {
57 std::vector<cudd_edge> out;
58 if (!h.ptr) return out;
59
60 if (Cudd_IsConstant(h.ptr)) return out;
61
62 // Handle possibly complemented node pointers. Use regular node to read
63 // children, then propagate the complement bit to the returned children
64 // so the rest of the code sees semantically-correct pointers.
65 const bool is_comp = Cudd_IsComplement(h.ptr);
66 DdNode* base = Cudd_Regular(h.ptr);
67
68 // else (0) then (1) ordering
69 DdNode* else_child = Cudd_E(base);
70 DdNode* then_child = Cudd_T(base);
71
72 if (is_comp) {
73 if (else_child) else_child = Cudd_Not(else_child);
74 if (then_child) then_child = Cudd_Not(then_child);
75 }
76
77 if (else_child) out.push_back(cudd_edge{handle{else_child}});
78 if (then_child) out.push_back(cudd_edge{handle{then_child}});
79
80 return out;
81 }
82
83 auto roots() const {
84 if (!mgr_ || roots_.empty()) return std::vector<handle>{};
85 std::vector<handle> out;
86 out.reserve(roots_.size());
87 std::transform(roots_.begin(), roots_.end(), std::back_inserter(out),
88 [](auto r) { return handle{r}; });
89 return out;
90 }

92 static auto start_guard(const handle&) { return dagir::noop_guard{}; }

94 private:
