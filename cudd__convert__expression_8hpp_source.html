31inline DdNode* convert_expression_to_cudd(DdManager& mgr, const dagir::utility::my_expression& expr,
32 std::unordered_map<std::string, int>& var_map) {
33 auto resolve_var = [&](const std::string& name) -> int {
34 auto it = var_map.find(name);
35 if (it != var_map.end()) return it->second;
36 int idx = static_cast<int>(var_map.size());
37 var_map.emplace(name, idx);
38 return idx;
39 };
